按钮组

// 创建按钮组
const buttonGroup = document.createElement('div');
buttonGroup.className = 'button-group';
buttonGroup.innerHTML = `
${UIComponents.renderButton('上一步', 'alert("上一步")', 'outline primary', 'medium', 'back')}
${UIComponents.renderButton('下一步', 'alert("下一步")', 'primary', 'medium', 'next')}
`;
document.getElementById('btn-group').appendChild(buttonGroup);
