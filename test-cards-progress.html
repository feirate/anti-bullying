反霸凌小英雄 - 卡片和进度条组件

这个页面展示了游戏中使用的卡片和进度条组件

基础卡片

// 基础卡片
UIComponents.renderCard(
'基础卡片标题',
'<p>这是卡片内容，可以包含任何HTML元素。</p>'
);

// 带图片的卡片
UIComponents.renderCard(
'带图片的卡片',
'<p>这是带图片的卡片内容。</p>',
'https://via.placeholder.com/300x150?text=示例图片'
);

带底部操作的卡片

// 带底部操作的卡片
UIComponents.renderCard(
'带底部操作的卡片',
'<p>这是卡片内容，底部有操作按钮。</p>',
'https://via.placeholder.com/300x150?text=示例图片',
UIComponents.renderButtonGroup([
{ text: '取消', onClick: 'alert("取消")', type: 'outline secondary' },
{ text: '确定', onClick: 'alert("确定")', type: 'primary' }
], 'flex-end')
);

场景卡片

// 场景卡片
const sampleScenario = {
id: 'sample_scenario',
title: '示例场景',

};

UIComponents.renderScenarioCard(sampleScenario, false);

基础进度条

// 基础进度条
UIComponents.renderProgressBar(3, 10);

// 带标签的进度条
UIComponents.renderProgressBar(7, 10, '场景进度');

彩色进度条

// 彩色进度条
UIComponents.renderProgressBar(3, 10, '初级进度', '#4A6FFF');
UIComponents.renderProgressBar(6, 10, '中级进度', '#FF9F1C');
UIComponents.renderProgressBar(9, 10, '高级进度', '#6BCF7F');
